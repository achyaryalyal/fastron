yum -y update && yum -y upgrade && cat /etc/os-release

yum -y install perl curl

iptables-save > ~/firewall.rules && systemctl stop firewalld.service && systemctl disable firewalld.service

nano /etc/selinux/config
# change SELINUX=enforcing to SELINUX=disabled
hostnamectl set-hostname server.[YOUR_DOMAIN]

swapon --show && fallocate -l 4G /swapfile && chmod 600 /swapfile && mkswap /swapfile && swapon /swapfile && swapon --show
nano /etc/fstab
# paste it at last line:
/swapfile swap swap defaults 0 0
# check again
swapon --show

cd /home && curl -o latest -L https://securedownloads.cpanel.net/latest && sh latest

service NetworkManager start && chkconfig NetworkManager on
reboot
